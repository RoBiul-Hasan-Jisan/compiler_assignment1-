// Enhanced Custom STL Library (Pure Logic Version)
class STLLibrary {
public:
    // Custom QuickSort implementation
    void sort(int arr[], int left, int right, bool ascending = true) {
        if (left >= right) return;

        int pivot = arr[(left + right) / 2];
        int i = left, j = right;

        while (i <= j) {
            if (ascending) {
                while (arr[i] < pivot) i++;
                while (arr[j] > pivot) j--;
            } else {
                while (arr[i] > pivot) i++;
                while (arr[j] < pivot) j--;
            }

            if (i <= j) {
                swap(arr[i], arr[j]);
                i++; j--;
            }
        }

        if (left < j) sort(arr, left, j, ascending);
        if (i < right) sort(arr, i, right, ascending);
    }

    // Swap two elements
    void swap(int &a, int &b) {
        int temp = a;
        a = b;
        b = temp;
    }

    // Reverse array in-place
    void reverse(int arr[], int size) {
        int left = 0, right = size - 1;
        while (left < right) {
            swap(arr[left], arr[right]);
            left++; right--;
        }
    }

    // Linear search
    int find(int arr[], int size, int target) {
        for (int i = 0; i < size; i++) {
            if (arr[i] == target) return i;
        }
        return -1; // Not found
    }

    // Deep copy
    void copy(int src[], int dest[], int size) {
        for (int i = 0; i < size; i++) {
            dest[i] = src[i];
        }
    }

    // Simulate malloc (return dynamically allocated array)
    int* malloc(int size) {
        int* ptr = new int[size];
        for (int i = 0; i < size; i++) ptr[i] = 0; // initialize with 0
        return ptr;
    }

    // Simulate free
    void free(int* ptr) {
        delete[] ptr;
    }

    // Custom printf (basic version)
    void printf(const char* format, int arg1 = 0, int arg2 = 0) {
        // Replace placeholders %1, %2 with args
        string out = format;
        size_t pos;
        pos = out.find("%1");
        if (pos != string::npos) out.replace(pos, 2, to_string(arg1));
        pos = out.find("%2");
        if (pos != string::npos) out.replace(pos, 2, to_string(arg2));

        cout << out << endl;
    }
};

int main() {
    STLLibrary stl;

    int arr[] = {50, 20, 30, 10, 40};
    int n = 5;

    // Sort
    stl.sort(arr, 0, n-1);
    stl.printf("Sorted array: %1 %2 %1 %2 %1", arr[0], arr[1]);  // basic printf demo

    // Reverse
    stl.reverse(arr, n);
    stl.printf("Reversed first two elements: %1 %2", arr[0], arr[1]);

    // Find
    int index = stl.find(arr, n, 30);
    stl.printf("Index of 30: %1", index);

    // Copy
    int arr_copy[5];
    stl.copy(arr, arr_copy, n);
    stl.printf("Copy first element: %1", arr_copy[0]);

    // Malloc & Free
    int* mem = stl.malloc(5);
    stl.free(mem);

    return 0;
}
